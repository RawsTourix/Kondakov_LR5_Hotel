121MenuObject::process(const Container& menu, const string main_label,
122 const string exit_label, int indent, bool one_off) {
123 int number = 0;
124 bool one_off_menu_inclusion = false;
125 do {
126 if (!main_label.empty()) cout << main_label << endl << endl;
127 show_menu(menu, indent);
128 cout << "0. " << exit_label << endl << endl << endl;
129
130 if (InputControl::input(number, "Пункт меню: ", 0, static_cast<int>(menu.size()))) {
131 if (one_off) { cout << endl; break; }
132 else { cout << endl << endl << endl; }
133 continue;
134 }
135
136 if (!one_off) cout << endl << endl;
137 if (number == 0) break;
138
139 try {
140 auto it = find_if(menu.begin(), menu.end(),
141 [number](MenuObject m) { return m.get_number() == number; });
142
143 if (it != menu.end()) {
144 one_off_menu_inclusion = it->get_one_off_menu_inclusion();
145 if (it->get_submenu().empty()) {
146 (*it).act();
147 if (one_off) break;
148 }
149 else {
150 process(it->get_submenu(), it->get_label(), "Назад.", indent, one_off);
151 if (one_off) break;
152 continue;
153 }
154 }
155 }
156 catch (const exception& e) {
157 cout << "Ошибка: " << e.what();
158 }
159
160 if (one_off) break;
161 cout << (one_off_menu_inclusion ? "\n\n" : "\n\n\n");
162 one_off_menu_inclusion = false;
163 } while (number != 0);
164}
